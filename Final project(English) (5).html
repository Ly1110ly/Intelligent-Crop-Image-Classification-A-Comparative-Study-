def load_and_preprocess_data(csv_path):
    """Load CSV file and preprocess data"""
    df = pd.read_csv(csv_path)
    
    # Handle mixed path formats (Windows/Linux compatibility)
    df['image_path'] = df['image_path'].str.replace('\\\\', '/', regex=True).str.replace('\\', '/', regex=False)
    
    # Separate features and labels
    X = df.drop(columns=['image_path', 'split', 'label'])  # Feature matrix
    y = df['label']  # Labels
    
    # Encode labels (convert strings to numerical values)
    le = LabelEncoder()
    y_encoded = le.fit_transform(y)
    y_categorical = to_categorical(y_encoded)
    
    # Split training and test sets
    train_mask = df['split'] == 'train'
    test_mask = df['split'] == 'test'
    
    X_train = X[train_mask]
    X_test = X[test_mask]
    y_train = y_categorical[train_mask]
    y_test = y_categorical[test_mask]
    
    # Feature standardization
    scaler = StandardScaler()
    X_train_scaled = scaler.fit_transform(X_train)
    X_test_scaled = scaler.transform(X_test)
    
    return X_train_scaled, X_test_scaled, y_train, y_test, le.classes_

# 2. Train and evaluate multiple models
def train_and_evaluate_models(X_train_scaled, X_test_scaled, y_train, y_test):
    """Train multiple models and evaluate performance"""
    if y_train.ndim > 1:
        y_train = np.argmax(y_train, axis=1)  # 
        y_test = np.argmax(y_test, axis=1)    #  [0,1,0] -> 1
        
    models = {
        'Logistic Regression': LogisticRegression(max_iter=1000, random_state=42),
        'Random Forest': RandomForestClassifier(n_estimators=100, random_state=42),
        'LightGBM': lgb.LGBMClassifier(num_leaves=31, learning_rate=0.05, n_estimators=100)
    }
    
    results = {}
    confusion_matrices = {}
    training_times = {}
    
    for name, model in models.items():
        start_time = time.time()
        
        # Train model(s)
        model.fit(X_train_scaled, y_train)
        
        # Prediction
        y_pred = model.predict(X_test_scaled)
        
        # Calculate training time
        training_time = time.time() - start_time
        training_times[name] = training_time
        model.predict(X_test_scaled[:1])# 
        n_repeats = 100  # 
        start_time_inference = time.time()
        for _ in range(n_repeats):
            model.predict(X_test_scaled)  # 
        total_inference_time = time.time() - start_time_inference
        # 
        n_samples = X_test_scaled.shape[0] * n_repeats
        single_image_latency = total_inference_time / n_samples  # 
        inference_speed = 1 / single_image_latency  # images/sec
        # ===============  ===============
        
        # Evaluate performance
        accuracy = accuracy_score(y_test, y_pred)
        report = classification_report(y_test, y_pred, output_dict=True)
        cm = confusion_matrix(y_test, y_pred)

        # precisionrecall
        precision = report['weighted avg']['precision']
        recall = report['weighted avg']['recall']
        f1 = report['weighted avg']['f1-score']
        
        results[name] = {
            'accuracy': accuracy,
            'precision': precision,
            'recall': recall,
            'f1':f1,
            'training_time': training_time,
            'single_image_latency': single_image_latency, 
            'inference_speed': inference_speed
        }
        
        confusion_matrices[name] = cm
        
        print(f"{name} Completed | "
      f"Accuracy: {accuracy:.4f} | "
      f"Training time: {training_time:.2f} seconds | "
      f"Precision: {precision:.4f} | "
      f"Recall: {recall:.4f} | "
      f"F1-score: {f1:.4f} | "
      f"Single Image Latency: {single_image_latency:.4f} seconds | "
      f"Inference Speed: {inference_speed:.2f} images/second")
    
    return results, confusion_matrices, training_times

def train_models_on_pca(models, X_train_pca, X_test_pca, y_train, y_test):

    confusion_matrices = {}
    training_times = {}
    
    for model_name, model in models.items():
        print(f"Training model: {model_name}...")
        start_time = time.time()
        
        # 
        model.fit(X_train_pca, y_train)
        training_time = time.time() - start_time
        
        # 
        y_pred = model.predict(X_test_pca)
        
        # ==========  ==========
        # 
        n_repeats = 100
        start_inference = time.time()
        for _ in range(n_repeats):
            model.predict(X_test_pca)  # 
        total_inference_time = time.time() - start_inference
        
        n_samples = X_test_pca.shape[0] * n_repeats
        single_image_latency = total_inference_time / n_samples
        inference_speed = 1 / single_image_latency
        
        # 
        pca_results[model_name] = {
            'accuracy': accuracy_score(y_test, y_pred),
            'precision': precision_score(y_test, y_pred, average='weighted'),
            'recall': recall_score(y_test, y_pred, average='weighted'),
            'f1': f1_score(y_test, y_pred, average='weighted'),
            'training_time': training_time,
            'single_image_latency': single_image_latency,
            'inference_speed': inference_speed
        }
        confusion_matrices[model_name] = confusion_matrix(y_test, y_pred)
        training_times[model_name] = training_time
    
    return pca_results, confusion_matrices, training_times

# 3. Visualize evaluation results
def visualize_results(results, confusion_matrices, training_times, class_names):
    """Visualize model evaluation results (supports dynamic layout)"""
    n_confusion = len(confusion_matrices)  # Retrieve confusion matrix counts
    ncols = 3  # Up to 2 subplots per row
    nrows = (n_confusion + 2) // ncols + 1  # Dynamically calculate rows: 1 row (performance + time) + N rows confusion matrix
    
    plt.figure(figsize=(15, 5 * nrows))  # Height dynamically adjusts based on row count
    
    # === 1. Model performance comparison ===
    plt.subplot(nrows, ncols, 1)  # Dynamic positioning
    metrics_df = pd.DataFrame(results).T
    metrics_df = metrics_df[['accuracy', 'precision', 'recall', 'f1']]
    metrics_df.plot(kind='bar', ax=plt.gca())
    plt.title('Model Performance Metrics Comparison')
    plt.ylabel('Score')
    plt.xticks(rotation=15)
    plt.legend(loc='lower right')
    plt.ylim(0, 1.05)
    
    # === 2. Training time comparison ===
    plt.subplot(nrows, ncols, 2)  # Dynamic positioning
    pd.Series(training_times).plot(kind='bar', color='orange')
    plt.title('Model Training Time Comparison')
    plt.ylabel('Time (seconds)')
    plt.xticks(rotation=15)

    # ===  ===
    plt.subplot(nrows, ncols, 3)
    inference_speeds = {name: res['inference_speed'] for name, res in results.items()}
    pd.Series(inference_speeds).plot(kind='bar', color='purple')
    plt.title('Model Inference Speed Comparison')
    plt.ylabel('Images/Second')
    plt.xticks(rotation=15)
    plt.grid(axis='y', linestyle='--')
    
    # === 3. Visualize confusion matrix (dynamic layout) ===
    for i, (name, cm) in enumerate(confusion_matrices.items(), start=4):
        plt.subplot(nrows, ncols, i)
        sns.heatmap(cm, annot=True, fmt='d', cmap='Blues',
                   xticklabels=class_names, yticklabels=class_names)
        plt.title(f'{name} Confusion Matrix')
        plt.xlabel('Predicted Labels')
        plt.ylabel('True Labels')
    
    plt.tight_layout()
    plt.savefig('model_comparison.png')
    plt.show()
    
    return metrics_df

def create_model(model_type):
    """"""
    if model_type == 'Logistic Regression':
        return LogisticRegression(max_iter=1000)
    else:
        return RandomForestClassifier(n_estimators=100)

def get_feature_importance(model, model_type):
    """"""
    if model_type == 'Logistic Regression':
        # 
        return np.mean(np.abs(model.coef_), axis=0)
    else:
        return model.feature_importances_

def plot_single_feature_importance(importances, title, feature_names=None, top_n=20):
    """"""
    # importances1D
    if len(importances.shape) > 1:
        importances = importances.flatten()
    
    idx = np.argsort(importances)[::-1][:top_n]
    sorted_importances = importances[idx]
    
    # 
    if feature_names is not None:
        sorted_labels = [feature_names[i] for i in idx]
    else:
        sorted_labels = [f'Feature {i}' for i in idx]
    
    plt.figure(figsize=(10, 8))
    plt.barh(range(len(idx)), sorted_importances, color='skyblue')
    plt.yticks(range(len(idx)), sorted_labels)
    plt.title(title)
    plt.xlabel('Importance Score')
    plt.gca().invert_yaxis()
    plt.tight_layout()
    plt.show()

def compare_feature_importance(X_train_scaled, y_train, X_train_pca_50, X_train_pca_95, 
                               pca_50, pca_95, model_names, feature_names=None):
    """PCA"""
    # ==========  ==========
    if len(X_train_scaled) != len(y_train):
        raise ValueError(f"The sample number of X_train_scaled and y_train is different: {len(X_train_scaled)} vs {len(y_train)}")
    if len(X_train_pca_50) != len(y_train):
        raise ValueError(f"The sample number of X_train_pca_50 and y_train is different: {len(X_train_pca_50)} vs {len(y_train)}")
    if len(X_train_pca_95) != len(y_train):
        raise ValueError(f"The sample number of X_train_pca_95 and y_train is different: {len(X_train_pca_95)} vs {len(y_train)}")
    
    pca_configs = {
        'PCA 50': (X_train_pca_50, pca_50),
        'PCA 95% Variance': (X_train_pca_95, pca_95)
    }
    all_importances = {}
    
    # ========== y_train ==========
    # y_train2Dn_samples, 1
    # y_train2Dn_samples, 1
    if len(y_train.shape) == 1:
        y_train = y_train.reshape(-1, 1)
    elif len(y_train.shape) > 1:
        y_train = y_train[:, 0].reshape(-1, 1)
    
    for model_name in model_names:
        print(f"\n{'='*30}")
        print(f"Feature importance analysis: {model_name}")
        print(f"{'='*30}")
        
        # ==========  ==========
        model = create_model(model_name)
        model.fit(X_train_scaled, y_train)
        original_imp = get_feature_importance(model, model_name)
        all_importances[f"{model_name}_Raw"] = original_imp
        
        # 
        plot_single_feature_importance(
            original_imp, 
            f'Feature importance of raw data ({model_name})',
            feature_names
        )
        
        # ========== PCA ==========
        for pca_name, (X_pca, pca_obj) in pca_configs.items():
            model = create_model(model_name)
            model.fit(X_pca, y_train)
            pca_imp = get_feature_importance(model, model_name)
            
            # ==========  ==========
            # 1
            n_features = X_train_scaled.shape  # 
            # 2
            pca_mapped_imp = np.zeros(n_features)
            
            # pca_imp1D
            pca_imp = pca_imp.flatten()
            
            # 
            # 3
            n_components = min(pca_obj.components_.shape, len(pca_imp))
            for i in range(n_components):
                # 4
                pca_mapped_imp += np.abs(pca_obj.components_[i]) * pca_imp[i]
            
            all_importances[f"{model_name}_{pca_name}"] = pca_mapped_imp
            
            # PCA
            plot_single_feature_importance(
                pca_mapped_imp, 
                f'{pca_name} PCA Mapped Feature Importance ({model_name})',
                feature_names
            )
    
    # 
    print("\n\n" + "="*60)
    print("Feature Importance Ranking Summary")
    print("="*60)
    
    for key, importances in all_importances.items():
        sorted_indices = np.argsort(importances)[::-1]
        
        # 
        if feature_names is not None:
            sorted_names = [feature_names[i] for i in sorted_indices]
        else:
            sorted_names = [f"Featrue_{i}" for i in sorted_indices]
        
        print(f"\n{key} - Top 10 Important Features:")
        for rank, idx in enumerate(sorted_indices[:10]):
            print(f"   {rank+1}. {sorted_names[rank]}: {importances[idx]:.4f}")
    
    return all_importances
